       ```typescript
       export async function registerDocResources(server: McpServer): Promise<void> {
         const docsDir = getDocsDir();
         const entries = await scanDirectory(docsDir, "L0");

         for (const entry of entries) {
           server.registerResource(
             entry.slug,
             `docs://androidcommondoc/${entry.slug}`,
             {
               title: titleFromSlug(entry.slug),
               description: entry.metadata.description ?? `Pattern documentation: ${titleFromSlug(entry.slug)}`,
               mimeType: "text/markdown",
             },
             async (uri) => {
               try {
                 const content = await readFile(entry.filepath, "utf-8");
                 return { contents: [{ uri: uri.href, text: content }] };
               } catch {
                 throw new McpError(ErrorCode.InvalidRequest, `Document not found: ${entry.slug}`);
               }
             },
           );
         }
       }
       ```

    - find-pattern({query: "testing", include_content: true}) includes full doc content in results
    - Rate-limited via shared limiter

    1. Create `mcp-server/src/tools/find-pattern.ts`:
       - Import scanDirectory from `../registry/scanner.js`
       - Import getDocsDir from `../utils/paths.js`
       - Import RegistryEntry, PatternMetadata from `../registry/types.js`
       - Import RateLimiter from `../utils/rate-limiter.js`

       - Import readFile from `node:fs/promises`
       - Import z from `zod`
       - Import logger from `../utils/logger.js`

       Zod input schema:
       ```typescript
       z.object({
         query: z.string().describe("Search term matching scope, sources, or targets metadata fields"),
         project: z.string().optional().describe('Project name for L1 resolution, or "all" for cross-project search. Omit for L0 only.'),
         targets: z.array(z.string()).optional().describe("Filter results to specific target platforms (e.g., ['android', 'desktop'])"),
         include_content: z.boolean().optional().default(false).describe("Include full document content in results (default: false for token efficiency)"),
       })
       ```

       Matching logic:
       - Tokenize query string by spaces and commas
       - For each token, check if it matches (case-insensitive substring) any value in entry's `scope`, `sources`, or `targets` arrays
       - An entry matches if ANY token matches ANY metadata field value
       - If `targets` filter provided, also require that entry.metadata.targets intersects with the filter

       Project resolution:
       - No project (undefined): scan L0 docs dir only
       - Specific project name: scan L0 + L2 + L1 for that project, resolve per-slug (highest priority wins)
       - "all": use discoverProjects() from project-discovery.ts, scan L0 + L2 + all L1 dirs, resolve per-slug per-project

       Response format (following existing tool response patterns):
       ```typescript
       {
         content: [{
           type: "text",
           text: JSON.stringify({
             query: "testing",
             matches: [
               {
                 slug: "testing-patterns",
                 description: "Standard patterns for testing...",
                 scope: ["testing", "coroutines", "quality"],
                 sources: ["kotlinx-coroutines-test", ...],
                 targets: ["android", "desktop", ...],
                 layer: "L0",
                 uri: "docs://androidcommondoc/testing-patterns",
                 content?: "..." // only if include_content=true
               }
             ],
             total: 3,
             project_filter: null
           })
         }]
       }
       ```
